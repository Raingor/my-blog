# 数据库授权

mysql 初始用户 root没有密码 需要 修改密码

mysql -uroot -p

进入mysql然后修改密码

set password for 'root' @ 'localhost' = password('123456');

修改完成退出:exit

远程授权 连接mysql

GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' IDENTIFIED BY 'mypassword' WITH GRANT OPTION;

FLUSH   PRIVILEGES;

1. create user  zx_root   IDENTIFIED by 'xxxxx';   //identified by 会将纯文本密码加密作为散列值存储
2. grant all privileges on [*|数据库名].* to '用戶'@'作用域' identified by '密碼'；

通过授权能直接新增用户
