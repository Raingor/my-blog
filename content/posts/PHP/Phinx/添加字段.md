# 添加字段

```bash
$table = $this->table('ctmon_ad_position');
      $table
        ->addColumn('ver','biginteger',array('limit' => 20,'signed'=>false,'null'=>false,'default' =>0,'comment' => '在校时长'))
        ->addColumn('create_time', 'integer',array('limit'  =>  10,'signed'=>false,'default'=>0,'comment'=>'创建时间'))
        ->addColumn('update_time', 'integer',array('limit'  =>  10,'signed'=>false,'default'=>0,'comment'=>'修改时间'))
        ->addColumn('delete_time', 'integer',array('limit'  =>  10,'signed'=>false,'default'=>0,'comment'=>'删除时间'))
        ->changeComment('广告定位表||@ver@create_time@update_time@delete_time')
        ->update();
```
