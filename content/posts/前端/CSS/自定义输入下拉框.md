# 自定义输入下拉框

在input标签下添加

html:
<div class="down-pop">  
    {volist name="proCat" id="vo"}  
        <div class="pop-item">1111</div>  
    {/volist}  
</div>

js:
```javascript
document.querySelectorAll('.down-pop').forEach((item,index)=>{  
  const temp_input=item.parentElement.querySelector('input')  
  temp_input.addEventListener('click',function(){  
    item.style.display='block';  
  })  
  item.addEventListener('click',function(e){  
    temp_input.value=e.target.innerText;  
    item.style.display='none';  
  })  
  temp_input.addEventListener('blur',function(){  
    setTimeout(() => {  
      item.style.display='none';  
    }, 500);    
  })  
    
})
```

css:

```bash
  
.down-pop{  
    position: absolute;  
    top: 100%;  
    left: 0;  
    width: 100%;  
    background-color: #fff;  
    box-shadow:0 0 4px #3636362b;  
    z-index: 1;  
    display: none;  
    max-height: 2rem;  
    overflow-y: auto;  
  
}  
.pop-item{  
    padding: 0.2rem;  
}  
.pop-item:hover{  
    background-color: #f9f9f9;  
    cursor: pointer;  
}
```
